83 void Benchmarker::Results::computeMetric(planning_interface::MotionPlanResponse &run, Run &metrics)
84 {
85  TrajectoryPtr traj;
86  if (metrics.success)
87  traj = std::make_shared<Trajectory>(*run.trajectory_);
88 
89  if (options.options & MetricOptions::WAYPOINTS)
90  metrics.metrics["waypoints"] = metrics.success ? int(traj->getNumWaypoints()) : int(0);
91 
92  if (options.options & MetricOptions::PATH && metrics.success)
93  metrics.path = traj->getMessage();
94 
95  if (options.options & MetricOptions::LENGTH)
96  metrics.metrics["length"] = metrics.success ? traj->getLength() : 0.0;
97 
98  if (options.options & MetricOptions::CORRECT)
99  metrics.metrics["correct"] = metrics.success ? traj->isCollisionFree(scene) : false;
100 
101  if (options.options & MetricOptions::CLEARANCE)
102  metrics.metrics["clearance"] = metrics.success ? std::get<0>(traj->getClearance(scene)) : 0.0;
103 
104  if (options.options & MetricOptions::SMOOTHNESS)
105  metrics.metrics["smoothness"] = metrics.success ? traj->getSmoothness() : 0.0;
106 
107  if (metric_callback)
108  metric_callback(run, metrics);
109 }

123 void Benchmarker::captureProgress(const std::map<std::string, Planner::ProgressProperty> &properties,
124  std::vector<std::map<std::string, std::string>> &progress, double rate)
125 {
126  ros::WallTime start = ros::WallTime::now();
127  while (true)
128  {
129  {
130  std::unique_lock<std::mutex> lock(solved_mutex_);
131  if (solved_)
132  return;
133 
134  std::map<std::string, std::string> data;
135  double time = (ros::WallTime::now() - start).toSec();
136 
137  data["time REAL"] = std::to_string(time);
138  for (const auto &property : properties)
139  data[property.first] = property.second();
140 
141  progress.emplace_back(data);
142  }
143 
144  // Sleep until the next update
145  std::this_thread::sleep_for(std::chrono::milliseconds(static_cast<long int>(rate * 1000)));
146  }
147 }

149 void Benchmarker::benchmark(const std::vector<BenchmarkOutputterPtr> &outputs, const Options &options)
150 {
151  unsigned int count = 0; // Completed queries.
152  const unsigned int total = requests_.size() * options.runs; // Total number of benchmarking queries.
153  std::shared_ptr<std::thread> thread; // Thread for querying progress properties.
154 
155  for (const auto &request : requests_)
156  {
157  // Extract all benchmarking request information
158  const auto &name = request.first;
159  const auto &scene = std::get<0>(request.second);
160  const auto &planner = std::get<1>(request.second);
161  const auto &builder = std::get<2>(request.second);
162  const auto &msg = builder->getRequest();
163  Results::ComputeMetricCallbackFn metric_callback;
164 
165  if (metric_callback_allocator_)
166  metric_callback = metric_callback_allocator_(request.second);
167 
168  // Execute pre-run step.
169  planner->preRun(scene, msg);
170 
171  Results results(name, scene, planner, builder, options, metric_callback);
172 
173  // Get all progress property names.
174  const auto &pp = planner->getProgressProperties(scene, msg);
175  if (not pp.empty())
176  {
177  for (const auto &property : pp)
178  results.properties.emplace_back(property.first);
179  results.properties.emplace_back("time REAL");
180  }
181 
182  for (unsigned int j = 0; j < options.runs || options.runs == 0; ++j)
183  {
184  ros::WallTime start = ros::WallTime::now();
185  solved_ = false;
186 
187  // Capture planner progress.
188  const auto &pp = planner->getProgressProperties(scene, msg);
189 
190  std::vector<std::map<std::string, std::string>> progress;
191  if (not pp.empty())
192  {
193  thread.reset(
194  new std::thread([&] { captureProgress(pp, progress, options.progress_update_rate); }));
195  }
196 
197  // Run planning.
198  planning_interface::MotionPlanResponse response = planner->plan(scene, msg);
199  double time = (ros::WallTime::now() - start).toSec();
200 
201  // Notify progress thread.
202  {
203  std::unique_lock<std::mutex> lock(solved_mutex_);
204  solved_ = true;
205  }
206 
207  if (thread)
208  thread->join();
209 
210  // Collate results.
211  auto &run = results.addRun(j, time, response);
212  run.progress = progress;
213 
214  // If runs == 0, run until time runs out.
215  if (options.runs == 0)
216  {
217  double time_remaining = msg.allowed_planning_time - time;
218  if (time_remaining < 0.)
219  break;
220  builder->setAllowedPlanningTime(time_remaining);
221  }
222 
223  RBX_INFO("BENCHMARKING: [ %u / %u ] Completed", ++count, total);
224  }
225 
226  results.finish = IO::getDate();
227 
228  for (const BenchmarkOutputterPtr &output : outputs)
229  output->dumpResult(results);
230  }
231 }
