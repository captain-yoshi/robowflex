92  # Currently limited to 1-DoF joints (e.g., prismatic, continuous, or

139  ## @brief Adds keyframes to animate a moveit_msgs::RobotTrajectoryMsg.

146  def animate_path(self, path_file, fps = 60., start = 30, reverse = False):
147  path = rv.utils.read_YAML_data(path_file)
148 
149  trajectory = path["joint_trajectory"]
150  names = trajectory["joint_names"]
151 
152  last_time = float(trajectory["points"][-1]["time_from_start"])
153  last_frame = start
154 
155  for point in trajectory["points"]:
156  time = float(point["time_from_start"])
157  if reverse:
158  time = last_time - time
159 
160  frame = start + time * fps
161  if frame > last_frame:
162  last_frame = frame
163 
164  for value, name in zip(point["positions"], names):
165  self.set_joint(name, value)
166  self.add_keyframe(name, frame)
167 
168  return last_frame
