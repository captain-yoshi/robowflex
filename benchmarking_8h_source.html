51  /** \brief A utility to benchmark many robowflex::Planner instances against different queries, specified
52  * by robowflex::Scene and robowflex::MotionRequestBuilder instances.
53  *
54  * For efficiency, be sure to specify in the Options which metrics you want computed. For example,
55  * clearance can take a while to compute. To add more metrics, or possibly metrics that are unique to your
56  * planning problems, add a custom function.
57  */

62  */
63  enum MetricOptions
64  {
65  WAYPOINTS = 1 << 0, ///< Number of waypoints in path.
66  PATH = 1 << 1, ///< The entire path.
67  CORRECT = 1 << 2, ///< Is the path correct (no collisions?).
68  LENGTH = 1 << 3, ///< Length of the path.
69  CLEARANCE = 1 << 4, ///< Clearance of path from obstacles.
70  SMOOTHNESS = 1 << 5, ///< Smoothness of path.
71  };
